    def reverseString(self, s: List[str]) -> None:
        """
        Do not return anything, modify s in-place instead.
        """
        #给star赋0
        start = 0 
        #给结尾赋值等于列表长度-1
        end = len(s) - 1
        #当star小于end时 执行下面循环语句 当star不小于end时 不做操作。
        while start < end:
        #进行交换值的操作
            s[start], s[end] = s[end], s[start]
            #start=start+1
            start += 1
            #end=end-1
            end -= 1
